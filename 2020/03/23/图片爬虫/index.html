图片爬虫

代码：

#!/usr/bin/python3
# -*- coding: utf-8 -*-
import requests, os, time
from tqdm import tqdm
from bs4 import BeautifulSoup

site = 'http://t66y.com/thread0806.php?fid=16'
header = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/81.0.4044.9 Safari/537.36"}

titles = []
links = []

def get_info():
    re = requests.get(site,headers=header)
    re.encoding='gbk'
    td = BeautifulSoup(re.text, "html.parser").find_all('td', class_='tal')
    for i in td:
        titles.append(i.h3.string)
        links.append('http://t66y.com/' + i.a['href'])

def download_img(title, link):
    if not os.path.exists(title):
        os.makedirs(title)
    re = requests.get(link,headers=header)
    re.encoding='gbk'
    div = BeautifulSoup(re.text, "html.parser").find_all('div', class_='tpc_content do_not_catch')
    if len(div) == 0:
        print('*****匹配到空值*****')
    else:
        img = BeautifulSoup(str(div[0]), "html.parser").find_all('img')
        pbar = tqdm(total=len(img))
        for i in img:
            file_name = title + '/' + str(img.index(i)) + '.jpg'
            if not os.path.exists(file_name):
                download_link = i['data-src']
                re = requests.get(download_link, headers=header)
                with open(file_name, 'wb') as f:
                    f.write(re.content)
            pbar.update(1)
            time.sleep(0.5)
        pbar.close()


get_info()
print(titles)
print(len(titles))
for i in range(1, len(titles)):
    print(i)
    print(titles[i])
    print('#' * 100)
    print('正在下载第'+str(i)+'个帖子：'+titles[i])
    download_img(titles[i], links[i])
    print('#' * 100 + '\n\n\n')
#    time.sleep(3)
print('下载完成！')
